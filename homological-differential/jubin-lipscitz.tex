\section{序論}
\paragraph{設定}
\(f\colon M\to N\)を\(C^\infty\)多様体間の\textbf{連続}写像とする．
この\(f\)に対して
\begin{description}
  \item[Whitney cone] \(C_f\coloneqq C(\Gamma_f,\Gamma_f)\subset T(M\times N)\)
  \item[conormal] \(\Lambda_f\coloneqq \muS(\kk_{\Gamma_{f}})\subset T^\ast(M\times N)\) 
\end{description}
とおく．

\paragraph{Whitney coneとconormalの性質}
\[\begin{cases*}
  C_f\colon&closed symmetric cone,\\
  \Lambda_f\colon&coisotropic closed symmetric cone.
\end{cases*}\]
また，\(f\)が\(C^1\)級なら
\[\begin{cases}
  C_f=T\Gamma_{f},\\
  \Lambda_f=(T\Gamma_{f})^{\perp}=T^{\ast}_{\Gamma_{f}}(M\times N).
\end{cases}\]

\paragraph{主定理}
\begin{THM}
  次は同値．
  \begin{enumerate}[(1)]
    \item \(f\)はLipscitz.
    \item \(C_f\cap(0_{M}\times TN)\subset 0_{MN}\).
    \item \(\Lambda_f\cap(T^{\ast}M\times 0_N)\subset 0^{\ast}_{MN}\).
  \end{enumerate}
\end{THM}

\paragraph{論文の構成}
\begin{enumerate}[1節]\setcounter{enumi}{1}
  \item \(\muS\)の復習
  \item 部分集合の\(\muS\)の性質，cone, strict cone の性質
  \item \(C_f\)の定義と性質．Liscitz, strict differentiability の定義，chain rule.
  \item \(\Lambda_f\)の定義と性質．核の畳み込みを用いて\(\muS\)の性質を\(f\)に拡張．
  \item 実数値関数のときを調べる．Vichery subdifferential との関係．
  \item 主定理の証明
  \item 部分多様体に拡張．locally Lipschitz, strict differentiable の判定．
\end{enumerate}

\subsubsection*{多様体}
\(f\colon M\to N\)を多様体の射とする．
\(\Lambda_f\coloneqq T_{\Gamma_{f}}^{\ast}(M\times N)\)とおく．
\[\Lambda\cong M\times_{N}T^{\ast}N\]が成り立つ．

\(A\subset T^{\ast}M\), \(B\subset T^{\ast}N\)に対し，
\begin{equation}
  f_{\pi}f_{d}^{-1}(A)=A\overset{a}{\circ}\Lambda_f,\quad
  f_{d}f_{\pi}^{-1}(B)=\Lambda_f\overset{a}{\circ}B
\end{equation}
が成り立つ．

\(f\)が\textbf{平滑} (smooth) すなわち沈めこみとなることと
\begin{equation}
  \Lambda_{f}\cap(0^{\ast}_{M}\times T^{\ast}N)\subset0^{\ast}_{M\times N}
\end{equation}
となることは同値である．より一般に，閉錐\(B\subset T^{\ast}N\)に対して
\begin{equation}

\end{equation}
