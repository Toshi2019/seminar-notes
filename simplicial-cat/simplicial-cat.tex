\begin{abstract}
    単体的圏の性質についてまとめる．
\end{abstract}
\section*{記号など}
\begin{itemize}
    \item 1点から成る集合を\(\{\pt\}\)で表す．
    \item 集合\(X\)の基数を\(\abs{X}\)とか\(\card{X}\)とか\(\Card{X}\)で表す．
    \item 圏\(\cat\)の始対象を\(\varnothing_\cat\)，終対象を\(\pt_\cat\)で表す．
    \item 圏\(\cat\)の対象\(X\)から\(Y\)への射の集合を\(\Hom_\cat(X,Y)\)や\(\cat(X,Y)\)で表す．
    \item 有限集合とその間の写像のなす圏を\(\finSet\)で表す．
\end{itemize}

\section{単体的圏}
\cite{KS06}を参考にした．
\begin{leftbar}
\begin{DFN}
    \textbf{単体圏} (simplex category)\footnotemark[1] \(\cSimp\)を次で定める．
    \begin{align*}
        \Ob(\cSimp)&\coloneqq\left\{\text{有限全順序集合}\right\},\\
        \Hom_{\cSimp}(\sigma,\tau)&\coloneqq\left\{
            u\colon\sigma\to\tau;\text{ \(u\)は順序を保つ写像}
        \right\}.
    \end{align*}
    射の合成は写像の合成で定める．

    \(\cSimp\)の部分圏\(\widetilde{\cSimp}\)を次で定める．
    \begin{align*}
        \Ob\left(\widetilde{\cSimp}\right)
        &\coloneqq\left\{
            \sigma\in\cSimp; \sigma\ne\varnothing
        \right\},\\
        \Hom_{\widetilde{\cSimp}}(\sigma,\tau)
        &\coloneqq\left\{
            u\in\Hom_{\cSimp}(\sigma,\tau);\text{ \(u\)は最大元と最小元を保つ}
        \right\}.
    \end{align*}
\end{DFN}
\end{leftbar}
\footnotetext[1]{
    \cite{KS06}では，\(\cSimp\)のことを 
    simplicial category（単体的圏）と呼んでいるが，
    \cite{Ri16}とか\cite{RV22}ではこう呼んでいたのでそちらに合わせる．
    \cite{Lu09}では別の概念を表すのに simplicial category を
    使っていたので，衝突しないようにする意図もある．
}

有限全順序集合\(\left[n,m\right]\)を\(
    \left\{k\in\zz; n\leqq k\leqq m\right\}
\)で定める．\([0,n]\)をたんに\([n]\)とかくことが多い．

以下単体圏の性質を述べる．

\begin{PRP}\label{PRP:half-full}
    うめこみ関手\(
        \iota\colon\cSimp\hookrightarrow\finSet
    \)は半充満かつ忠実である．
\end{PRP}

\begin{RMK}
    関手\(F\colon\cat\to\cat'\)が\textbf{半充満} (half-full) であるとは，
    任意の対象\(X,Y\in\cat\)に対し，
    \(F(X)\cong F(Y)\)ならば，\(X\)と\(Y\)の間の同型が存在することをいう．
    ただし，\(\cat'\)における同型\(F(X)\simar F(Y)\)は
    \(\cat\)における同型\(X\to Y\)から来るものでなくともよい．    
    
    \(\cat\)の部分圏\(\cat'\)が半充満であるとは，
    うめこみ関手\(\cat\hookrightarrow\cat'\)が半充満であることをいう．
\end{RMK}

\begin{proof}[命題\ref{PRP:half-full}の証明]
    まず\(\iota\)が半充満であることを示す．
    \(\sigma, \tau\in\cSimp\)を
    \(\iota(\sigma)\cong\iota(\tau)\)をみたすものとすると，
    \(\card{\sigma}=\card{\tau}\)である．
\end{proof}
